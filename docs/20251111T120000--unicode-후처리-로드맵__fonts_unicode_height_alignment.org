def fix_korean_bearing(font_path, verbose=False):
    """í•œê¸€ ê¸€ë¦¬í”„ bearing ì¬ì¡°ì •"""
    font = fontforge.open(font_path)
    half_width = font[0x0030].width
    target_width = half_width * 2

    for glyph in font.glyphs():
        if (0xAC00 <= glyph.unicode <= 0xD7A3 or
            0x3131 <= glyph.unicode <= 0x318E):
            if glyph.width > half_width:
                bbox = glyph.boundingBox()
                actual_width = bbox[2] - bbox[0]
                offset = (target_width - actual_width) / 2 - bbox[0]
                glyph.transform(psMat.translate(offset, 0))
                glyph.width = target_width

    font.generate(font_path)
    font.close()

def fix_math_height(font_path, verbose=False):
    """Math symbols ë†’ì´ ì •ê·œí™”

    ëª©í‘œ:
    - ascent â‰¤ EM_ASCENT (880)
    - descent â‰¤ EM_DESCENT (120)
    - ë¹„ìœ¨ ìœ ì§€í•˜ë©´ì„œ ì¶•ì†Œ
    """
    font = fontforge.open(font_path)
    EM_ASCENT = 880
    EM_DESCENT = 120

    math_ranges = [
        (0x2200, 0x22FF, "Mathematical Operators"),
        (0x2A00, 0x2AFF, "Supplemental Mathematical Operators"),
        (0x27C0, 0x27EF, "Misc Math Symbols-A"),
        (0x2980, 0x29FF, "Misc Math Symbols-B"),
    ]

    fixed_count = 0

    for start, end, name in math_ranges:
        for code in range(start, end + 1):
            if code not in font:
                continue

            glyph = font[code]
            bbox = glyph.boundingBox()

            # bbox: (xmin, ymin, xmax, ymax)
            glyph_ascent = bbox[3]
            glyph_descent = abs(bbox[1]) if bbox[1] < 0 else 0

            # ë†’ì´ ì´ˆê³¼ ì—¬ë¶€ í™•ì¸
            needs_scaling = False
            scale_y = 1.0

            if glyph_ascent > EM_ASCENT:
                scale_y = min(scale_y, EM_ASCENT / glyph_ascent)
                needs_scaling = True

            if glyph_descent > EM_DESCENT:
                scale_y = min(scale_y, EM_DESCENT / glyph_descent)
                needs_scaling = True

            if needs_scaling:
                # ì¤‘ì‹¬ ê¸°ì¤€ìœ¼ë¡œ Yì¶• ìŠ¤ì¼€ì¼ë§
                center_y = (bbox[3] + bbox[1]) / 2
                glyph.transform(psMat.translate(0, -center_y))
                glyph.transform(psMat.scale(1, scale_y))
                glyph.transform(psMat.translate(0, center_y))
                fixed_count += 1

                if verbose:
                    print(f"  U+{code:04X} ({name}): scale_y={scale_y:.2f}")

    font.generate(font_path)
    font.close()

    return fixed_count

def fix_emoji_size(font_path, verbose=False):
    """Emoji í¬ê¸° ì¡°ì •

    ëª©í‘œ:
    - width â‰¤ full_width (1056 or 1000)
    - height â‰¤ EM_ASCENT + EM_DESCENT (1000)
    - ì¢…íš¡ë¹„ ìœ ì§€
    """
    font = fontforge.open(font_path)
    half_width = font[0x0030].width
    full_width = half_width * 2
    EM_ASCENT = 880
    EM_DESCENT = 120
    MAX_HEIGHT = EM_ASCENT + EM_DESCENT

    emoji_ranges = [
        (0x1F300, 0x1F5FF, "Misc Symbols and Pictographs"),
        (0x1F600, 0x1F64F, "Emoticons"),
        (0x1F680, 0x1F6FF, "Transport and Map"),
        (0x1F900, 0x1F9FF, "Supplemental Symbols"),
        (0x2600, 0x26FF, "Miscellaneous Symbols"),
    ]

    fixed_count = 0

    for start, end, name in emoji_ranges:
        for code in range(start, end + 1):
            if code not in font:
                continue

            glyph = font[code]
            bbox = glyph.boundingBox()

            actual_width = bbox[2] - bbox[0]
            actual_height = bbox[3] - bbox[1]

            # í¬ê¸° ì´ˆê³¼ ì—¬ë¶€ í™•ì¸
            scale = 1.0

            if actual_width > full_width:
                scale = min(scale, full_width / actual_width)

            if actual_height > MAX_HEIGHT:
                scale = min(scale, MAX_HEIGHT / actual_height)

            if scale < 1.0:
                # ì¤‘ì‹¬ ê¸°ì¤€ìœ¼ë¡œ uniform ìŠ¤ì¼€ì¼ë§
                center_x = (bbox[2] + bbox[0]) / 2
                center_y = (bbox[3] + bbox[1]) / 2
                glyph.transform(psMat.translate(-center_x, -center_y))
                glyph.transform(psMat.scale(scale, scale))
                glyph.transform(psMat.translate(center_x, center_y))

                # ì „ê° í­ ì„¤ì •
                glyph.width = full_width
                fixed_count += 1

                if verbose:
                    print(f"  U+{code:04X} ({name}): scale={scale:.2f}")

    font.generate(font_path)
    font.close()

    return fixed_count

import fontforge
import sys

def analyze_height(font_path):
    font = fontforge.open(font_path)
    EM_ASCENT = 880
    EM_DESCENT = 120

    violations = []

    for glyph in font.glyphs():
        if glyph.unicode == -1:
            continue

        bbox = glyph.boundingBox()
        ascent = bbox[3]
        descent = abs(bbox[1]) if bbox[1] < 0 else 0

        if ascent > EM_ASCENT or descent > EM_DESCENT:
            violations.append({
                'code': glyph.unicode,
                'char': chr(glyph.unicode),
                'ascent': ascent,
                'descent': descent,

                  f"ascent={v['ascent']:.0f} (limit={EM_ASCENT}), "
                  f"descent={v['descent']:.0f} (limit={EM_DESCENT})")
    else:
        print("âœ… ëª¨ë“  ê¸€ë¦¬í”„ê°€ ë†’ì´ ì œí•œ ì´ë‚´")

    font.close()
    return len(violations)
